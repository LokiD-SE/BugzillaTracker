import requests
import time
from datetime import datetime, timedelta

# --- CONFIGURATION ---
BUGZILLA_URL = "https://bugzilla.bizom.in/rest/bug"
GOOGLE_CHAT_WEBHOOK = "https://chat.googleapis.com/v1/spaces/AAAAu1J2E2g/messages?key=AIzaSyDdI0hCZtE6vySjMm-WEfRq3CPzqKqqsHI&token=3jFB5fa7aLRZaffagHDft5Bpj_BKSbj2eyaV0VE0g0g"
CHECK_INTERVAL_MINUTES = 60

# Your specific filters from the URL
QUERY_PARAMS = {
    "bug_status": ["UNCONFIRMED", "CONFIRMED", "IN_PROGRESS", "RESOLVED", "RE-OPENED"],
    "product": ["BizomWeb", "Mobile App"],
    "email1": "lokesh.devamuthu@mobisy.com",
    "emailtype1": "substring",
    "emailassigned_to1": 1,
    "last_change_time": "" # Will be updated dynamically
}

def check_bugzilla():
    # Only look for bugs changed since the last check
    since_time = (datetime.utcnow() - timedelta(minutes=CHECK_INTERVAL_MINUTES)).strftime('%Y-%m-%dT%H:%M:%SZ')
    QUERY_PARAMS["last_change_time"] = since_time
    
    try:
        response = requests.get(BUGZILLA_URL, params=QUERY_PARAMS)
        bugs = response.json().get('bugs', [])
        
        for bug in bugs:
            send_to_google_chat(bug)
    except Exception as e:
        print(f"Error checking Bugzilla: {e}")

def send_to_google_chat(bug):
    text = (
        f"üêû *Bug Updated: #{bug['id']}*\n"
        f"*Summary:* {bug['summary']}\n"
        f"*Status:* {bug['status']} | *Component:* {bug['component']}\n"
        f"üîó https://bugzilla.bizom.in/show_bug.cgi?id={bug['id']}"
    )
    
    payload = {"text": text}
    requests.post(GOOGLE_CHAT_WEBHOOK, json=payload)

if __name__ == "__main__":
    print("Monitoring Bugzilla...")
    while True:
        check_bugzilla()
        time.sleep(CHECK_INTERVAL_MINUTES * 60)
